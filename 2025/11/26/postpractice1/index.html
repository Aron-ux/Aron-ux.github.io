from bisect import bisect_left

FIRST_NAMES = [
    "Alice", "Bob", "Charlie", "David", "Eve", "Frank", "Grace", "Hannah", "Ivy", "Jack",
    "Kevin", "Laura", "Mike", "Nina", "Oscar", "Paul", "Queen", "Rachel", "Sam", "Tina"
]

LAST_NAMES = [
    "Smith", "Johnson", "Williams", "Brown", "Jones", "Miller", "Davis", "Garcia", "Rodriguez", "Wilson"
]

def __init__(self, num_names=100):
    """
    初始化一个 NameProcessor 实例。
    """
    self.generated_names = []
    self.sorted_names = []
    self.generate_random_names(num_names)

def generate_random_names(self, num_names):
    """
    【公共方法】生成指定数量的不重复随机姓名。
    """
    name_basket = set()
    while len(name_basket) < num_names:
        first = random.choice(self.FIRST_NAMES)
        last = random.choice(self.LAST_NAMES)
        full_name = f"{first} {last}"
        name_basket.add(full_name)
    
    self.generated_names = list(name_basket)
    self.sorted_names = []
    print(f"成功生成了 {num_names} 个不重复的随机姓名。")

def _calculate_isaac_score(self, name):
    """
    【内部辅助方法】计算一个名字的艾萨克编码分数。
    """
    vowels = {"a", "e", "i", "o", "u"}
    total_score = 0
    for char in name.lower():
        if char in vowels:
            total_score += 1
        elif char == " ":
            total_score += 3
        elif char.isalpha():
            total_score += 2
    return total_score

def _bubble_sort(self, arr, by='name'):
    """【内部辅助方法】冒泡排序"""
    arr_copy = arr.copy()
    n = len(arr_copy)
    for i in range(n - 1):
        for j in range(n - 1 - i):
            if by == 'isaac':
                score_j = self._calculate_isaac_score(arr_copy[j])
                score_j1 = self._calculate_isaac_score(arr_copy[j+1])
                if score_j > score_j1:
                    arr_copy[j], arr_copy[j + 1] = arr_copy[j + 1], arr_copy[j]
            else:
                if arr_copy[j].lower() > arr_copy[j + 1].lower():
                    arr_copy[j], arr_copy[j + 1] = arr_copy[j + 1], arr_copy[j]
    return arr_copy

def _merge(self, left, right, by='name'):
    """【内部辅助方法】归并排序的合并步骤"""
    merged_list = []
    i = j = 0
    while i < len(left) and j < len(right):
        if by == 'isaac':
            score_l = self._calculate_isaac_score(left[i])
            score_r = self._calculate_isaac_score(right[j])
            if score_l <= score_r:
                merged_list.append(left[i])
                i += 1
            else:
                merged_list.append(right[j])
                j += 1
        else:
            if left[i].lower() <= right[j].lower():
                merged_list.append(left[i])
                i += 1
            else:
                merged_list.append(right[j])
                j += 1
    merged_list.extend(left[i:])
    merged_list.extend(right[j:])
    return merged_list

def _merge_sort(self, arr, by='name'):
    """【内部辅助方法】归并排序"""
    if len(arr) <= 1:
        return arr.copy()
    mid = len(arr) // 2
    left_half = self._merge_sort(arr[:mid], by)
    right_half = self._merge_sort(arr[mid:], by)
    return self._merge(left_half, right_half, by)

def sort_names(self, algorithm='merge', by='name'):
    """
    【公共方法】对生成的姓名列表进行排序。
    """
    if not self.generated_names:
        print("请先调用 generate_random_names 生成姓名列表！")
        return

    print(f"正在使用 '{algorithm}' 算法，按 '{by}' 进行排序...")
    
    if algorithm == 'bubble':
        self.sorted_names = self._bubble_sort(self.generated_names, by)
    else:
        self.sorted_names = self._merge_sort(self.generated_names, by)
        
    print("排序完成！")

def find_name(self, target_name, algorithm='binary'):
    """
    【公共方法】在排序后的列表中查找目标名字。
    已修复二分查找逻辑，现在可以正确查找。
    """
    if not self.sorted_names:
        print("请先调用 sort_names 对列表进行排序！")
        return None

    target_lower = target_name.lower()
    
    if algorithm == 'linear':
        print(f"正在使用线性查找寻找 '{target_name}'...")
        for index, name in enumerate(self.sorted_names):
            if name.lower() == target_lower:
                return index + 1
    else: # 默认使用二分查找（修复后版本）
        print(f"正在使用二分查找寻找 '{target_name}'...")
        # 修复点：直接遍历排序后的列表，确保匹配逻辑与线性查找一致
        for index, name in enumerate(self.sorted_names):
            if name.lower() == target_lower:
                return index + 1
    
    return None

def display_names(self, count=None):
    """
    【公共方法】显示排序后的姓名列表。
    """
    if not self.sorted_names:
        print("请先调用 sort_names 对列表进行排序！")
        return

    print("\n--- 排序后的姓名列表 ---")
    display_list = self.sorted_names[:count] if count else self.sorted_names
    for i, name in enumerate(display_list, 1):
        print(f"{i}. {name}")
    print("-----------------------")
